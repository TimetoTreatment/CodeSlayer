void BubbleSort(int ar[], int n)
{
    int i, j, temp;

    for (i = n - 1; i > 0; i--)
    {
        for (j = 0; j < i; j++)
        {
            if (ar[j] > ar[j + 1])
            {
                 temp = ar[j];
                 ar[j] = ar[j + 1];
                 ar[j + 1] = temp;
            }
        }
    }
}
