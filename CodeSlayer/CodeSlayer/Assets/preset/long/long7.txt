int isPrime(int n)
{
    int i;
    int sqroot;

    for (i = 2;; i++)    
    {
        if (i * i > n)   
        {
            sqroot = i - 1;
            break;      
        }
    }

    for (i = 2; i <= sqroot; i++)
    {
        if (n % i == 0)         
            return 0;   
    }

    return 1;
}
